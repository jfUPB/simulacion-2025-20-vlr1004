class Walker {
  constructor() {
    this.x = width / 2;
    this.y = height / 2;
  }

  step() {
    let angle = random(TWO_PI);

    let stepSize = this.levy();
    
    this.x += cos(angle) * stepSize;
    this.y += sin(angle) * stepSize;

    this.x = constrain(this.x, 0, width);
    this.y = constrain(this.y, 0, height);
  }

  levy() {
    let r = random(1);
    return 1 / pow(r, 1.5);
  }

  show() {
    stroke(0);
    strokeWeight(2);
    point(this.x, this.y);
  }
}
